**1. Restructure server initialization (top of file)**

With explicit http.createServer to support the health endpoint:
```javascript
import http from "http";

const httpServer = http.createServer((req, res) => {
  if (req.method === "GET" && req.url === "/health") {
    const health = {
      status: "ok",
      uptime: Math.floor(process.uptime()),
      timestamp: new Date().toISOString(),
      rooms: rooms.map((r) => ({
        id: r.id,
        name: r.name,
        players: r.characters.length,
        bots: r.characters.filter((c) => c.isBot).length,
      })),
      totalPlayers: rooms.reduce((sum, r) => sum + r.characters.length, 0),
      totalBots: rooms.reduce(
        (sum, r) => sum + r.characters.filter((c) => c.isBot).length,
        0
      ),
    };
    res.writeHead(200, { "Content-Type": "application/json" });
    res.end(JSON.stringify(health));
    return;
  }
  // Non-health requests: return 404 (Socket.IO attaches its own listener)
  res.writeHead(404);
  res.end();
});

const io = new Server(httpServer, {
  cors: { origin },
});
```

Change `io.listen(3000)` to `httpServer.listen(3000)` at the same location (keep it near the top, before the console.log).

**3. Add emote:play handler and ALLOWED_EMOTES constant**

Add the ALLOWED_EMOTES constant near the top of the file (after imports, before pathfinding):
```javascript
const ALLOWED_EMOTES = ["dance", "wave", "sit", "nod"];
```

Inside the `io.on("connection")` handler, add the `emote:play` event AFTER the existing `dance` handler (keep the dance handler for backward compatibility):
```javascript
socket.on("emote:play", (emoteName) => {
  if (!room) return;
  if (typeof emoteName !== "string") return;
  if (!ALLOWED_EMOTES.includes(emoteName)) return;
  io.to(room.id).emit("emote:play", {
    id: socket.id,
    emote: emoteName,
  });
});
```

IMPORTANT: Keep the existing `dance` handler unchanged — the current client listens for `playerDance`, not `emote:play`. Removing it breaks the dance button. Migration happens in Phase 6.

IMPORTANT: Do NOT pass a callback to `http.createServer()` AND also call `httpServer.on("request")` — use only the callback approach to avoid double-handling (see Pitfall 5 in RESEARCH.md).

IMPORTANT: The health endpoint callback references `rooms` which is defined later in the file. This is fine because the callback only runs when an HTTP request arrives, and rooms is populated by `loadRooms()` before the server starts listening. But make sure `httpServer.listen(3000)` is called AFTER `loadRooms()` in the file flow. Currently `io.listen(3000)` is at line 12 (before loadRooms at line 90). Move `httpServer.listen(3000)` to after `loadRooms()` or keep it where it is since JavaScript hoisting and the async nature of HTTP requests means rooms will be populated before any request handler fires. The safest approach: keep `httpServer.listen(3000)` right after the `new Server(httpServer, ...)` call, same location as the old `io.listen(3000)`. The rooms array is initialized as `[]` at line 58 and populated synchronously by `loadRooms()` at line 90. Since `httpServer.listen()` is async (it sets up the listener and returns), and `loadRooms()` runs synchronously after, rooms will be populated before any real request arrives.

```javascript
import { io } from "socket.io-client";

const socket = io("http://localhost:3000");
let passed = 0;
let failed = 0;

socket.on("welcome", (data) => {
  console.log("Connected. Rooms:", data.rooms.length);
  socket.emit("joinRoom", data.rooms[0].id, {
    avatarUrl: "https://models.readyplayer.me/test.glb",
    isBot: true,
    name: "TestBot",
  });
});

socket.on("roomJoined", (data) => {
  const me = data.characters.find((c) => c.id === data.id);

  // Check isBot flag
  if (me.isBot === true) { passed++; console.log("PASS: isBot is true"); }
  else { failed++; console.log("FAIL: isBot is", me.isBot); }

  // Check name field
  if (me.name === "TestBot") { passed++; console.log("PASS: name is TestBot"); }
  else { failed++; console.log("FAIL: name is", me.name); }

  // Test emote:play
  socket.emit("emote:play", "wave");
});

socket.on("emote:play", (data) => {
  if (data.emote === "wave") { passed++; console.log("PASS: emote:play wave received"); }
  else { failed++; console.log("FAIL: emote data", data); }

  // Test invalid emote (should be silently rejected)
  socket.emit("emote:play", "invalid_emote");

  // Give time for invalid emote to NOT arrive, then check health
  setTimeout(async () => {
    const res = await fetch("http://localhost:3000/health");
    const health = await res.json();
    if (health.status === "ok" && health.totalBots === 1) {
      passed++; console.log("PASS: health shows 1 bot");
    } else {
      failed++; console.log("FAIL: health", health);
    }

    console.log(`\nResults: ${passed} passed, ${failed} failed`);
    socket.disconnect();
    process.exit(failed > 0 ? 1 : 0);
  }, 1000);
});
```
